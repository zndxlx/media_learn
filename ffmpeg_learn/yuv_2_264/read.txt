#将一段mp4提取yuv数据
#ss 5 提取视频开始时间
#-t 20  提取视频时长
#-pix_fmt yuv420p  指定yuv为yuv420p格式
#-s 1280x720   指定编码分辨率
ffmpeg -ss 5 -t 20 -i input.mp4 -s 1280x720 -pix_fmt yuv420p 1280x720_yuv420p.yuv
